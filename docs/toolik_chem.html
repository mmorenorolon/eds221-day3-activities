#subset from the “Toolik Inlet” site, and that contains the variable for pH, dissolved organic carbon (DOC), and total dissolved nitrogen (TDN)
inlet_biochem <- subset(toolik_biochem, site == "Toolik Inlet") %>% select(p_h, doc_u_m, tdn_u_m)
view(inlet_biochem)
#Write a for loop from scratch to calculate the mean for each column
for (i in seq_along(inlet_biochem)) { #go through each column (seq_along)
  mean_value <- mean(inlet_biochem[[i]], na.rm = TRUE) #access numeric vectors 
  # [[]] for each column
  print(paste(names(inlet_biochem)[i], ":", mean_value)) #names() to grab the 
  # column names for a variable, followed by a colon, and the result of the mean 
  # value for that variable
}
